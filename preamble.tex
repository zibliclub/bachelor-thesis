% --- Figure and table captions per requirements ----------------------------
% Figure: caption below, centered, format "Рис. N. Title".
% Table:  caption above, right-aligned, 12pt, format "Таблица N. Title".
\captionsetup[figure]{
  name=Рис.,
  labelsep=period,
  justification=centering,
  singlelinecheck=true,
  position=below
}
\captionsetup[table]{
  name=Таблица,
  labelsep=period,
  justification=raggedleft,
  singlelinecheck=false,
  font=small,
  position=above
}

% --- Heading style ---------------------------------------------------------
% Chapters (\section): 16pt; sections and subsections: 14pt.
% All headings: bold, left-aligned with paragraph indent, no trailing period.
\usepackage{titlesec}
\titleformat{\section}
  {\normalfont\fontsize{16pt}{19pt}\bfseries}
  {\thesection.}{0.5em}{}
\titleformat{\subsection}
  {\normalfont\fontsize{14pt}{17pt}\bfseries}
  {\thesubsection.}{0.5em}{}
\titleformat{\subsubsection}
  {\normalfont\fontsize{14pt}{17pt}\bfseries}
  {\thesubsubsection.}{0.5em}{}

% Unnumbered chapter-level headings — Introduction, Conclusion, References,
% Appendices, Table of Contents. New page, paragraph indent, left-aligned, 16pt bold.
\newcommand{\thesisChapterStar}[1]{%
  \clearpage
  \phantomsection
  \addcontentsline{toc}{section}{#1}%
  \noindent\hspace*{\parindent}\textbf{\fontsize{16pt}{19pt}\selectfont #1}\par
  \vspace{\baselineskip}%
}

% --- Table of contents -----------------------------------------------------
% Look inspired by vcourse-document-template:
%   - section: bold name + bold page number;
%   - subsection: bold number, regular name and page;
%   - uniform dot leaders on every level.
% The "Содержание" heading itself follows ВКР formatting rules: capitalized first letter,
% left-aligned with paragraph indent, bold, no trailing period.
\usepackage{tocloft}

\renewcommand{\cftsecfont}{\bfseries}
\renewcommand{\cftsecpagefont}{\bfseries}

\renewcommand{\cftsubsecfont}{\mdseries}
\renewcommand{\cftsubsecpresnum}{\bfseries}
\renewcommand{\cftsubsecaftersnum}{\mdseries}
\renewcommand{\cftsubsecpagefont}{\mdseries}

\renewcommand{\cftdotsep}{2}
\renewcommand{\cftsecdotsep}{\cftdotsep}
\renewcommand{\cftsecleader}{\cftdotfill{\cftsecdotsep}}

\setlength{\cftbeforesecskip}{4pt}

\renewcommand{\contentsname}{Содержание}

% Print the heading ourselves (in \thesisChapterStar style) and call
% \@starttoc directly so tocloft does not emit its own title.
\makeatletter
\newcommand{\makethesistoc}{%
  \clearpage
  \thispagestyle{fancy}%
  \noindent\hspace*{\parindent}\textbf{\fontsize{16pt}{19pt}\selectfont \contentsname}\par
  \vspace{\baselineskip}%
  \@starttoc{toc}%
  \clearpage
}
\makeatother

% --- Appendices ------------------------------------------------------------
% Russian-letter appendices per requirements.md: letters А, Б, В, Г, Д, Е, Ж,
% И, К, Л, М, Н, П, Р, С, Т, У, Ф, Х, Ц, Ш, Щ, Э, Ю, Я (Ё, З, Й, О, Ч, Ь, Ы, Ъ
% are skipped). Letters are passed explicitly to avoid auto-numbering mistakes.
%
% Usage in a section file:
%   \thesisAppendix{А}{Скриншоты iOS-приложения}
% Renders as a new-page, left-aligned, bold heading
% «Приложение А. Скриншоты iOS-приложения» and adds an entry to the ToC.
\newcommand{\thesisAppendix}[2]{%
  \clearpage
  \phantomsection
  \addcontentsline{toc}{section}{Приложение #1. #2}%
  \noindent\hspace*{\parindent}%
    \textbf{\fontsize{16pt}{19pt}\selectfont Приложение #1. #2}\par
  \vspace{\baselineskip}%
}
