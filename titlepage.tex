% Title page following the OmSU (Dostoevsky Omsk State University) template.
% All values are sourced from meta.tex.

\RequirePackage{etoolbox}
\newcommand{\thesisProgramLine}{%
  \ifdefstring{\thethesisProgramKind}{specialty}%
    {по специальности \thethesisProgramCode\,--\,\thethesisProgramName}%
    {по направлению \thethesisProgramCode\,--\,\thethesisProgramName}%
}

\newcommand{\thesisSignatureLine}{%
  \rule{2.7cm}{0.4pt}%
}

\newcommand{\thesisDateLine}{%
  «\rule{0.8cm}{0.4pt}»~\rule{2.5cm}{0.4pt}~\thethesisYear~г.%
}

% Signature blocks. They are rendered left-aligned but share one common width
% (\thesisBlockWidth, the widest of the three) so their left edges line up while
% the whole stack still hugs the right margin.
\newlength{\thesisBlockWidth}

\newcommand{\thesisApprovedBlock}{%
  \begin{tabular}{@{}l@{}}%
    УТВЕРЖДАЮ\\
    Заведующий кафедрой\\[2pt]
    \thesisSignatureLine~\thethesisHeadName\\[2pt]
    \thesisDateLine
  \end{tabular}%
}

\newcommand{\thesisAdvisorBlock}{%
  \begin{tabular}{@{}l@{}}%
    Научный руководитель:\\
    \thethesisAdvisorRank\\[2pt]
    \thesisSignatureLine~\thethesisAdvisorName\\[2pt]
    \thesisDateLine
  \end{tabular}%
}

\newcommand{\thesisStudentBlock}{%
  \begin{tabular}{@{}l@{}}%
    Выполнил:\\
    студент группы \thethesisStudentGroup\\[2pt]
    \thesisSignatureLine~\thethesisStudentName\\[2pt]
    \thesisDateLine
  \end{tabular}%
}

% Measure all three blocks and keep the widest width.
\newcommand{\thesiscomputeblockwidth}{%
  \sbox0{\thesisApprovedBlock}%
  \sbox2{\thesisAdvisorBlock}%
  \sbox4{\thesisStudentBlock}%
  \setlength{\thesisBlockWidth}{\wd0}%
  \ifdim\wd2>\thesisBlockWidth \setlength{\thesisBlockWidth}{\wd2}\fi
  \ifdim\wd4>\thesisBlockWidth \setlength{\thesisBlockWidth}{\wd4}\fi
}

% Render one block left-aligned inside the shared width, flush against the right margin.
\newcommand{\thesissignatureblock}[1]{%
  \begin{flushright}%
    \makebox[\thesisBlockWidth][l]{#1}%
  \end{flushright}%
}

\newcommand{\makethesistitle}{%
  \begin{titlepage}%
    \thispagestyle{thesistitle}%
    \begin{singlespace}%
    \setlength{\parindent}{0pt}%
    \thesiscomputeblockwidth
    \begin{center}
      \thethesisMinistry\\[2pt]
      \thethesisInstitutionLong\\[2pt]
      \thethesisUniversity\\[2pt]
      \thethesisDepartment
    \end{center}

    \vspace{12mm}

    \thesissignatureblock{\thesisApprovedBlock}

    \vspace{20mm}

    % Thesis title
    \begin{center}
      {\fontsize{16pt}{20pt}\selectfont \thethesisTitle\par}

      \vspace{10mm}

      Выпускная квалификационная работа\\
      \thesisProgramLine
    \end{center}

    \vspace{18mm}

    \thesissignatureblock{\thesisAdvisorBlock}

    \vspace{10mm}

    \thesissignatureblock{\thesisStudentBlock}

    \vfill

    \begin{center}
      \thethesisCity\\
      \thethesisYear
    \end{center}
    \end{singlespace}%
  \end{titlepage}%
  % Title page counts as page 1 but its number is not printed.
  \setcounter{page}{2}%
}
